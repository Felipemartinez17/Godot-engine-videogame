class_name MainMenu
extends Control

@onready var start_button = $MarginContainer/HBoxContainer/VBoxContainer/Start_Button as Button
@onready var info_button = $MarginContainer/HBoxContainer/VBoxContainer/Info_Button as Button
@onready var exit_button = $MarginContainer/HBoxContainer/VBoxContainer/Exit_Button2 as Button
@onready var level = preload("res://Scene/level.tscn") as PackedScene

func _ready() -> void:
	start_button.button_down.connect(_on_start_pressed)
	exit_button.button_down.connect(_on_exit_pressed)
	
func _on_start_pressed() -> void:
	get_tree().change_scene_to_packed(level)
	
func _on_exit_pressed() -> void:
	get_tree().quit()
